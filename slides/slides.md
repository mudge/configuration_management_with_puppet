# A Resource

      uid        => '507',
      gid        => 'admin',
      shell      => '/bin/zsh',
      home       => '/home/henry',

# Directory Structure

    @@@ ruby
    file { '/home/myapp/apps':
        ensure => directory,
        owner => 'myapp',
        group => 'myapp',
        require => User['myapp'],
    }

# Apache Configuration

    @@@ ruby
    file {
      '/etc/httpd/sites_available/myapp.conf':
        ensure => present,
        content => template("myapp.conf");

      '/etc/httpd/sites_enabled/myapp.conf':
        ensure => link,
        target => '/etc/httpd/sites_available/myapp.conf';
    }
